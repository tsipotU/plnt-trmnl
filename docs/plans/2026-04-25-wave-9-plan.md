```tsx
// ErrorBoundary.tsx
import { Component, ReactNode, ErrorInfo } from 'react';

interface Props { children: ReactNode }
interface State { hasError: boolean; error: Error | null }

export class ErrorBoundary extends Component<Props, State> {
  state: State = { hasError: false, error: null };

  static getDerivedStateFromError(error: Error): State {
    return { hasError: true, error };
  }

  componentDidCatch(error: Error, info: ErrorInfo) {
    console.error('[ErrorBoundary]', error, info.componentStack);
  }

  handleReload = () => window.location.reload();

  render() {
    if (this.state.hasError) {
      return (
        <div role="alert" style={{ padding: 24, textAlign: 'center' }}>
          <div style={{ fontSize: 48, marginBottom: 16 }}>🪴</div>
          <h1 style={{ fontSize: 22, fontWeight: 700, marginBottom: 8 }}>
            Something went wrong
          </h1>
          <p style={{ color: 'var(--text-secondary)', marginBottom: 24 }}>
            The app hit an unexpected error. Reloading often helps.
          </p>
          <button
            onClick={this.handleReload}
            style={{
              background: 'var(--accent)', color: 'white', border: 'none',
              padding: '12px 24px', borderRadius: 12, fontSize: 16, fontWeight: 600,
            }}
          >
            Reload
          </button>
          {this.state.error && (
            <details style={{ marginTop: 24, fontSize: 12, textAlign: 'left' }}>
              <summary style={{ cursor: 'pointer', color: 'var(--text-secondary)' }}>
                Technical details
              </summary>
              <pre style={{ overflow: 'auto', padding: 8 }}>
                {this.state.error.message}
              </pre>
            </details>
          )}
        </div>
      );
    }
    return this.props.children;
  }
}
```

```ts
// auth-state.ts
import { Router } from 'express';
import type Database from 'better-sqlite3';

export function createSystemRouter(db: Database.Database): Router {
  const router = Router();
  router.get('/ai-connection', (_req, res) => {
    const row = db.prepare(`
      SELECT 1 FROM event_log
      WHERE event_type = 'enrichment_callback'
        AND created_at >= datetime('now', '-7 days')
      LIMIT 1
    `).get();
    res.json({ connected: !!row });
  });
  return router;
}
```

```ts
const ROUNDS = 12;

export async function hashPassword(plain: string): Promise<string> {
```

- [ ] **Step 4: Implement sessions.ts**

```ts
import { randomUUID } from 'node:crypto';
import type Database from 'better-sqlite3';

const SESSION_TTL_DAYS = 30;

export function createSession(db: Database.Database): string {
  const id = randomUUID();
  db.prepare(`
    INSERT INTO sessions (id, expires_at)
    VALUES (?, datetime('now', '+${SESSION_TTL_DAYS} days'))
  `).run(id);
  return id;
}

export function validateSession(db: Database.Database, id: string): boolean {
  const row = db.prepare(`
    SELECT 1 FROM sessions WHERE id = ? AND expires_at > datetime('now')
  `).get(id);
  if (row) {
    db.prepare(`UPDATE sessions SET last_used_at = datetime('now') WHERE id = ?`).run(id);
    return true;
  }
  return false;
}

export function destroySession(db: Database.Database, id: string): void {
  db.prepare(`DELETE FROM sessions WHERE id = ?`).run(id);
}
```

```ts
import { Router } from 'express';
import { randomBytes } from 'node:crypto';
import type Database from 'better-sqlite3';
import { hashPassword, verifyPassword } from '../auth/passwords.js';
import { createSession, destroySession } from '../auth/sessions.js';

export function createAuthRouter(db: Database.Database): Router {
  const router = Router();

  function getAdminHash(): string | null {
    const row = db.prepare(`SELECT value FROM app_state WHERE key = 'admin_password_hash'`).get() as { value?: string } | undefined;
    return row?.value ?? null;
  }

  function getSetupToken(): string {
    const row = db.prepare(`SELECT value FROM app_state WHERE key = 'setup_token'`).get() as { value?: string } | undefined;
    if (row?.value) return row.value;
    const token = randomBytes(8).toString('hex').toUpperCase().match(/.{4}/g)!.join('-');
    db.prepare(`INSERT INTO app_state (key, value) VALUES ('setup_token', ?)`).run(token);
    console.log(`[auth] No admin password set — setup token: ${token}`);
    console.log(`[auth] Visit / and enter this token to claim this instance.`);
    return token;
  }

  router.get('/setup-token', (_req, res) => {
    if (getAdminHash()) {
      res.status(404).json({ error: 'Setup already complete' });
      return;
    }
    res.json({ token: getSetupToken() });
  });

  router.post('/setup', async (req, res) => {
    if (getAdminHash()) {
      res.status(403).json({ error: 'Setup already complete' });
      return;
    }
    const { token, password } = req.body ?? {};
    const expected = getSetupToken();
    if (token !== expected) { res.status(401).json({ error: 'Invalid setup token' }); return; }
    if (typeof password !== 'string' || password.length < 12) {
      res.status(400).json({ error: 'Password must be at least 12 characters' });
      return;
    }
    const hash = await hashPassword(password);
    db.prepare(`INSERT INTO app_state (key, value) VALUES ('admin_password_hash', ?)`).run(hash);
    db.prepare(`DELETE FROM app_state WHERE key = 'setup_token'`).run();
    const session = createSession(db);
    res.cookie('session', session, sessionCookieOptions(req)).json({ ok: true });
  });

  router.post('/login', async (req, res) => {
    const hash = getAdminHash();
    if (!hash) { res.status(403).json({ error: 'Setup not complete' }); return; }
    const { password } = req.body ?? {};
    if (typeof password !== 'string') { res.status(400).json({ error: 'password required' }); return; }
    if (!(await verifyPassword(password, hash))) {
      res.status(401).json({ error: 'Incorrect password' });
      return;
    }
    const session = createSession(db);
    res.cookie('session', session, sessionCookieOptions(req)).json({ ok: true });
  });

  router.post('/logout', (req, res) => {
    const session = req.cookies?.session;
    if (session) destroySession(db, session);
    res.clearCookie('session').json({ ok: true });
  });

  router.get('/me', (req, res) => {
    const session = req.cookies?.session;
    if (!session) { res.status(401).json({ authenticated: false }); return; }
    res.json({ authenticated: true });
  });

  return router;
}

function sessionCookieOptions(req: any) {
  return {
    httpOnly: true,
    sameSite: 'strict' as const,
    secure: req.secure || req.headers['x-forwarded-proto'] === 'https',
    maxAge: 30 * 24 * 60 * 60 * 1000,
  };
}
```

```ts
// auth/middleware.ts
import type { Request, Response, NextFunction } from 'express';
import type Database from 'better-sqlite3';
import { validateSession } from './sessions.js';

const PUBLIC_PREFIXES = ['/health', '/api/auth/', '/api/feedback']; // feedback unauthenticated for the dog-food window — revisit
const PUBLIC_EXACT = new Set(['/']);

export function requireAuth(db: Database.Database) {
  return function (req: Request, res: Response, next: NextFunction) {
    if (PUBLIC_PREFIXES.some((p) => req.path.startsWith(p))) return next();
    if (PUBLIC_EXACT.has(req.path)) return next();

    // Bootstrap mode — no admin password set, let SPA bootstrap handle setup
    const hashRow = db.prepare(`SELECT value FROM app_state WHERE key = 'admin_password_hash'`).get();
    if (!hashRow) return next();

    const session = req.cookies?.session;
    if (!session || !validateSession(db, session)) {
      res.status(401).json({ error: 'Authentication required' });
      return;
    }
    next();
  };
}
```

Wrap routes with a guard component that:

2. If 401 + no admin set → redirect to `/setup`
3. If 401 + admin set → redirect to `/login`

```tsx
function AuthGate({ children }: { children: React.ReactNode }) {
  const [state, setState] = useState<'loading' | 'auth' | 'setup' | 'login'>('loading');
  const location = useLocation();
  useEffect(() => {
    if (location.pathname === '/setup' || location.pathname === '/login') {
      setState('auth'); return;
    }
    fetch('/api/auth/me').then(async (r) => {
      if (r.ok) { setState('auth'); return; }
      // 401 — figure out which: try setup-token endpoint
      const tokRes = await fetch('/api/auth/setup-token');
      setState(tokRes.ok ? 'setup' : 'login');
    });
  }, [location.pathname]);
  if (state === 'loading') return <div style={{padding:24,textAlign:'center'}}>Loading…</div>;
  if (state === 'setup') return <Navigate to="/setup" replace />;
  if (state === 'login') return <Navigate to={`/login?next=${encodeURIComponent(location.pathname)}`} replace />;
  return <>{children}</>;
}
```
